const db = window.supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY); let subjectsData = []; let activeSubject = null; let chartObj = null; // DOM Elements const listEl = document.getElementById("subjectList"); const ttList = document.getElementById("ttList"); const uiTotal = document.getElementById("inpTotal"); const uiAttended = document.getElementById("inpAttended"); const uiTarget = document.getElementById("inpTarget"); const uiSkips = document.getElementById("inpSkips"); async function fetchInitialData() { const { data: { session } } = await db.auth.getSession(); if (!session) { window.location.href = "student-signin.html"; return; } try { // Fetch basic attendance const { data: att } = await db.from('attendance_summary').select('*').eq('student_id', session.user.id); // Fetch subjects to map names const { data: subs } = await db.from('subjects').select('id, subject_name, subject_code'); if (!subs || subs.length === 0) { listEl.innerHTML = `
No subjects found in Timetable. Seed DB.
`; return; } const attMap = {}; if (att) { att.forEach(a => { attMap[a.subject_id] = { total: parseInt(a.total_classes) || 0, attended: parseInt(a.attended) || 0, pct: parseFloat(a.percentage) || 0 }; }); } subjectsData = subs.map(s => { const a = attMap[s.id] || { total: 0, attended: 0, pct: 0 }; return { id: s.id, name: s.subject_name.substring(0, 20), code: s.subject_code, total: a.total, attended: a.attended, pct: a.pct }; }); // Sort by attendance percentage subjectsData.sort((a, b) => a.pct - b.pct); renderSubjects(); if (subjectsData.length > 0) { selectSubject(subjectsData[0].id); } } catch (e) { console.error(e); listEl.innerHTML = `
Error syncing data.
`; } } function renderSubjects() { listEl.innerHTML = subjectsData.map(s => { let col = "text-safe"; if (s.pct < 75) col = "text-warn"; if (s.pct < 65) col = "text-danger"; return `
${s.name}
${s.code}
${s.pct}%
`; }).join(''); } async function selectSubject(id) { // Update active states document.querySelectorAll('.subject-item').forEach(el => el.classList.remove('active')); document.getElementById(`sub-${id}`).classList.add('active'); activeSubject = subjectsData.find(s => s.id === id); // Populate Details document.getElementById("activeSubName").textContent = activeSubject.name; document.getElementById("activeSubCode").textContent = activeSubject.code; uiTotal.value = activeSubject.total; uiAttended.value = activeSubject.attended; uiSkips.value = "0"; calculatePrediction(); await fetchTimetable(); } function getStatusInfo(pct, target) { if (pct >= target) return { cl: "text-safe", bg: "bg-safe", text: "SAFE", risk: "No immediate action needed" }; if (pct >= target - 5) return { cl: "text-warn", bg: "bg-warn", text: "WARNING", risk: "Approaching danger zone" }; return { cl: "text-danger", bg: "bg-danger", text: "AT RISK", risk: "Immediate attendance required" }; } function calculatePrediction() { if (!activeSubject) return; const t = parseInt(uiTotal.value) || 0; const a = parseInt(uiAttended.value) || 0; const target = parseInt(uiTarget.value) || 75; const skips = parseInt(uiSkips.value) || 0; const targetDec = target / 100; const currentPct = t > 0 ? (a / t) * 100 : 0; const sInfo = getStatusInfo(currentPct, target); // 1. Current Stats Updates document.getElementById("currentPct").textContent = currentPct.toFixed(1) + "%"; document.getElementById("currentPct").className = `text-4xl outfit font-black ${sInfo.cl}`; const b = document.getElementById("currentBadge"); b.className = `absolute top-4 right-4 text-[10px] font-bold px-2 py-1 rounded-full uppercase tracking-wider ${sInfo.bg} ${sInfo.cl}`; b.textContent = sInfo.text; b.classList.remove('hidden'); // 2. Target Logistics const actionMsg = document.getElementById("actionMsg"); const statusMsg = document.getElementById("statusMsg"); if (currentPct >= target) { // Buffer calc const bufferClasses = Math.floor(a * (100 / target) - t); statusMsg.innerHTML = ` Eligible`; actionMsg.innerHTML = `You can safely skip ${Math.max(0, bufferClasses)} more classes.`; } else { // Required calc: (target * total - attended) / (1 - target) const reqClasses = Math.ceil((targetDec * t - a) / (1 - targetDec)); statusMsg.innerHTML = ` Not Eligible`; actionMsg.innerHTML = `You MUST attend ${Math.max(0, reqClasses)} consecutive classes.`; } // 3. Sim Result Display const simEl = document.getElementById("simResult"); if (skips > 0) { const newT = t + skips; const newPct = (a / newT) * 100; const nInfo = getStatusInfo(newPct, target); simEl.classList.remove("hidden"); simEl.className = `p-4 rounded-xl flex items-center gap-4 mt-2 ${nInfo.bg} ${nInfo.cl}`; document.getElementById("simCount").textContent = skips; document.getElementById("simPct").textContent = newPct.toFixed(1) + "%"; document.getElementById("simRisk").innerHTML = nInfo.text === "SAFE" ? " Still Safe" : " Puts you at Risk!"; } else { simEl.classList.add("hidden"); } // 4. Update Chart updateChart(t, a, skips, target); } function updateChart(total, attended, skips, target) { const ctx = document.getElementById('trendChart').getContext('2d'); if (chartObj) chartObj.destroy(); // Next 10 future classes const labels = Array.from({ length: 10 }, (_, i) => `Class +${i + 1}`); // Baseline: If student attends all next 10 classes const baselineData = Array.from({ length: 10 }, (_, i) => { const futTotal = total + i + 1; const futAttended = attended + i + 1; return (futAttended / futTotal) * 100; }); // Target Line const targetData = Array(10).fill(target); const datasets = [ { label: "If you attend all", data: baselineData, borderColor: "#34d399", // Emerald backgroundColor: "rgba(52, 211, 153, 0.1)", borderWidth: 2, tension: 0.4, fill: true, pointRadius: 3 }, { label: "Target Required", data: targetData, borderColor: "rgba(229, 56, 59, 0.6)", borderWidth: 2, borderDash: [5, 5], pointRadius: 0, fill: false } ]; if (skips > 0) { const skipData = Array.from({ length: 10 }, (_, i) => { const futTotal = total + i + 1; // For the next `skips` classes, attended doesn't increase. // After `skips` classes, assume they attend. const extraAttended = i < skips ? 0 : (i + 1 - skips); const futAttended = attended + extraAttended; return (futAttended / futTotal) * 100; }); datasets.push({ label: `With ${skips} skips`, data: skipData, borderColor: "#fbbf24", // Amber backgroundColor: "rgba(251, 191, 36, 0.1)", borderWidth: 2, tension: 0.4, fill: true, pointRadius: 3 }); } chartObj = new Chart(ctx, { type: 'line', data: { labels, datasets }, options: { responsive: true, maintainAspectRatio: false, color: "#94a3b8", plugins: { legend: { position: 'top', labels: { boxWidth: 10, font: { family: 'Inter', size: 10 } } }, tooltip: { mode: 'index', intersect: false, backgroundColor: 'rgba(0,0,0,0.8)' } }, scales: { x: { grid: { color: "rgba(255,255,255,0.05)" }, ticks: { font: { size: 9 } } }, y: { grid: { color: "rgba(255,255,255,0.05)" }, min: Math.max(0, Math.min(...baselineData) - 15), max: 100, ticks: { callback: v => v + "%" } } } } }); } async function fetchTimetable() { ttList.innerHTML = `
Checking Timetable...
`; try { const DAY_WEIGHT = { "Monday": 1, "Tuesday": 2, "Wednesday": 3, "Thursday": 4, "Friday": 5 }; const { data: slots, error } = await db.from("timetable").select("*").eq("subject_id", activeSubject.id); if (error || !slots || slots.length === 0) { ttList.innerHTML = `
No timetable slots found for this subject.
`; return; } // Sort by day then time slots.sort((a, b) => (DAY_WEIGHT[a.day] - DAY_WEIGHT[b.day]) || a.start_time.localeCompare(b.start_time)); ttList.innerHTML = slots.map(s => { const isToday = s.day === new Date().toLocaleDateString('en-US', { weekday: 'long' }); const badge = isToday ? 'Today' : ''; return `
${s.day.substring(0, 3)}
${s.start_time} - ${s.end_time} ${badge}
${s.room}
`; }).join(''); } catch (e) { console.error(e); ttList.innerHTML = `
Failed to fetch slots.
`; } } // Event Listeners [uiTarget, uiSkips].forEach(el => el.addEventListener('input', calculatePrediction)); // Start fetchInitialData();
